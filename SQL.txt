CREATE TABLE CART 
(
 CARTID INTEGER primary key generated always as identity,
 PRODUCTCODE VARCHAR(255),
 PRODUCTNAME VARCHAR(255),
 COLOR VARCHAR(255), 
 QTY INTEGER, 
 PRICE DOUBLE, 
 TOTALPRICE DOUBLE, 
 IMG VARCHAR(255),
 CUSTOMERID INTEGER
);
